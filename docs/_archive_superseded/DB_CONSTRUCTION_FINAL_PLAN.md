### 통합된 내용

- ✅ 에러 핸들링 개선

### 🛠 기술 스택

- **Backend**: FastAPI + Supabase Python Client (Service Role)
- **Frontend**: Next.js 15 + Supabase JS Client + @supabase/ssr

## 1. 데이터베이스 스키마 (SQL)

### 1.1 Phase 1 + Phase 2 기본 테이블

Supabase SQL Editor에서 실행할 최종 스크립트입니다.

```sql
-- ============================================
-- CR-Check 데이터베이스 스키마 v2.0
-- Phase 1-2: 기본 테이블 + 사용자 프로필
-- ============================================

BEGIN;

-- 1. Users Profile (사용자 정보 확장)
-- 목적: auth.users의 메타데이터 확장, RLS 정책 적용 가능
CREATE TABLE public.profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  email TEXT NOT NULL,
  nickname TEXT NOT NULL DEFAULT split_part(email, '@', 1),
  role TEXT DEFAULT 'user' CHECK (role IN ('user', 'admin')),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,

  -- 닉네임 중복 방지 (검색/표시 시 혼란 방지)
  CONSTRAINT unique_nickname UNIQUE (nickname)
);

-- 인덱스
CREATE INDEX idx_profiles_nickname ON public.profiles(nickname);

-- RLS: 프로필
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

CREATE POLICY "공개 프로필은 누구나 조회 가능"
  ON public.profiles FOR SELECT
  USING (true);

CREATE POLICY "사용자는 자신의 프로필 생성 가능"
  ON public.profiles FOR INSERT
  WITH CHECK (auth.uid() = id);

CREATE POLICY "사용자는 자신의 프로필만 수정 가능"
  ON public.profiles FOR UPDATE
  USING (auth.uid() = id);

COMMENT ON TABLE public.profiles IS '사용자 프로필 (auth.users 확장)';
COMMENT ON COLUMN public.profiles.nickname IS '표시 이름 (기본값: 이메일 앞부분, 중복 불가)';


-- Trigger: 회원가입 시 자동으로 public.profiles에 행 생성
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.profiles (id, email, nickname)
  VALUES (
    new.id,
    new.email,
    COALESCE(new.raw_user_meta_data->>'full_name', split_part(new.email, '@', 1))
  )
  ON CONFLICT (nickname) DO UPDATE SET nickname = EXCLUDED.nickname || '_' || substr(md5(random()::text), 1, 4);
  -- 닉네임 중복 시 랜덤 접미사 추가
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();

COMMENT ON FUNCTION public.handle_new_user() IS '회원가입 시 프로필 자동 생성 (닉네임 중복 방지 포함)';


-- 2. Articles (기사 원본 메타데이터)
-- 목적: 중복 분석 방지, 언론사별 통계 기반
CREATE TABLE public.articles (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  url TEXT NOT NULL,
  title TEXT NOT NULL,
  publisher TEXT,
  journalist TEXT,
  publish_date TIMESTAMP WITH TIME ZONE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,

  CONSTRAINT articles_url_key UNIQUE (url)
);

-- 인덱스
CREATE INDEX idx_articles_url ON public.articles(url);
CREATE INDEX idx_articles_publisher ON public.articles(publisher);
CREATE INDEX idx_articles_created_at ON public.articles(created_at DESC);

-- RLS: Articles
ALTER TABLE public.articles ENABLE ROW LEVEL SECURITY;

CREATE POLICY "기사는 누구나 조회 가능"
  ON public.articles FOR SELECT
  USING (true);
-- 쓰기는 Service Role(백엔드)만 가능 (RLS 우회)

COMMENT ON TABLE public.articles IS '기사 원본 메타데이터 (중복 방지용)';
COMMENT ON COLUMN public.articles.url IS '기사 고유 식별자 (중복 분석 방지)';


-- 3. Analysis Results (분석 결과)
-- 목적: AI 분석 결과 저장, 사용자 연결 (Phase 2)
CREATE TABLE public.analysis_results (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  article_id BIGINT REFERENCES public.articles(id) ON DELETE CASCADE NOT NULL,
  user_id UUID REFERENCES public.profiles(id) ON DELETE SET NULL, -- 비로그인 가능

  -- 리포트 내용
  comprehensive_report TEXT,
  journalist_report TEXT,
  student_report TEXT,

  -- 메타데이터
  model_version TEXT DEFAULT 'claude-sonnet-4.5',
  phase1_model TEXT DEFAULT 'claude-haiku-4.5',
  phase2_model TEXT DEFAULT 'claude-sonnet-4.5',
  duration_seconds FLOAT,
  detected_categories JSONB, -- Phase 1 결과 (예: ["1-1-1", "1-7-3"])

  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 인덱스
CREATE INDEX idx_analysis_article ON public.analysis_results(article_id);
CREATE INDEX idx_analysis_user ON public.analysis_results(user_id);
CREATE INDEX idx_analysis_created_at ON public.analysis_results(created_at DESC);

-- RLS: Analysis Results
ALTER TABLE public.analysis_results ENABLE ROW LEVEL SECURITY;

CREATE POLICY "분석 결과는 누구나 조회 가능"
  ON public.analysis_results FOR SELECT
  USING (true);
-- 쓰기는 Service Role(백엔드)만 가능

COMMENT ON TABLE public.analysis_results IS 'AI 분석 결과 저장';
COMMENT ON COLUMN public.analysis_results.detected_categories IS 'Phase 1에서 탐지된 문제 카테고리 ID 목록';

COMMIT;
```

### 1.2 Phase 3 통계 함수

Phase 3 시작 시 추가로 실행할 SQL 함수들입니다.

```sql
-- ============================================
-- CR-Check 통계 함수
-- Phase 3: 인사이트 대시보드
-- ============================================

BEGIN;

-- 1. 실시간 트렌드 (최근 N시간 내 인기 기사)
CREATE OR REPLACE FUNCTION get_trending_articles(hours_ago INT DEFAULT 24)
RETURNS TABLE (
  article_id BIGINT,
  title TEXT,
  url TEXT,
  publisher TEXT,
  analysis_count BIGINT,
  latest_analysis TIMESTAMP WITH TIME ZONE
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    a.id as article_id,
    a.title,
    a.url,
    a.publisher,
    COUNT(ar.id) as analysis_count,
    MAX(ar.created_at) as latest_analysis
  FROM public.articles a
  JOIN public.analysis_results ar ON a.id = ar.article_id
  WHERE ar.created_at >= NOW() - INTERVAL '1 hour' * hours_ago
  GROUP BY a.id, a.title, a.url, a.publisher
  ORDER BY analysis_count DESC, latest_analysis DESC
  LIMIT 10;
END;
$$ LANGUAGE plpgsql STABLE;

COMMENT ON FUNCTION get_trending_articles(INT) IS '최근 N시간 내 가장 많이 분석된 기사 TOP 10';


-- 2. 언론사별 통계
CREATE OR REPLACE FUNCTION get_publisher_stats()
RETURNS TABLE (
  publisher TEXT,
  total_analyses BIGINT,
  avg_duration FLOAT
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    a.publisher,
    COUNT(ar.id) as total_analyses,
    AVG(ar.duration_seconds) as avg_duration
  FROM public.articles a
  JOIN public.analysis_results ar ON a.id = ar.article_id
  WHERE a.publisher IS NOT NULL
  GROUP BY a.publisher
  ORDER BY total_analyses DESC
  LIMIT 20;
END;
$$ LANGUAGE plpgsql STABLE;

COMMENT ON FUNCTION get_publisher_stats() IS '언론사별 분석 횟수 및 평균 소요 시간 TOP 20';


-- 3. 전체 통계 요약
CREATE OR REPLACE FUNCTION get_overall_stats()
RETURNS TABLE (
  total_articles BIGINT,
  total_analyses BIGINT,
  avg_duration FLOAT,
  unique_users BIGINT
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    COUNT(DISTINCT a.id) as total_articles,
    COUNT(ar.id) as total_analyses,
    AVG(ar.duration_seconds) as avg_duration,
    COUNT(DISTINCT ar.user_id) FILTER (WHERE ar.user_id IS NOT NULL) as unique_users
  FROM public.articles a
  LEFT JOIN public.analysis_results ar ON a.id = ar.article_id;
END;
$$ LANGUAGE plpgsql STABLE;

COMMENT ON FUNCTION get_overall_stats() IS '전체 통계 요약 (총 기사, 분석, 사용자 수)';

COMMIT;
```

### 2.3 DatabaseManager 클래스 (개선판)

**파일 생성**: `backend/database.py`

```python
# backend/database.py
"""
Supabase 데이터베이스 연동 모듈

역할:
- 분석 결과를 Supabase에 저장
- 기사 중복 방지 (URL 기반)
- 사용자 연결 (Phase 2)
- 에러 핸들링 및 로깅
"""

import os
import logging
from datetime import datetime
from supabase import create_client, Client
from typing import Optional, Dict, Any

# 로깅 설정
logger = logging.getLogger(__name__)

class DatabaseManager:
    """분석 결과를 Supabase에 저장하는 관리자"""

    def __init__(self):
        """Supabase 클라이언트 초기화"""
        url = os.environ.get("SUPABASE_URL")
        key = os.environ.get("SUPABASE_SERVICE_ROLE_KEY")

        if not url or not key:
            logger.warning("⚠️ Supabase credentials missing. Database features disabled.")
            self.client = None
        else:
            try:
                self.client: Client = create_client(url, key)
                logger.info("✅ Supabase connected successfully")
            except Exception as e:
                logger.error(f"❌ Supabase connection failed: {e}")
                self.client = None

    async def save_or_get_article(self, article_data: Dict[str, Any]) -> Optional[int]:
        """
        기사 중복 확인 후 ID 반환

        Args:
            article_data: {
                "url": 기사 URL (필수),
                "title": 제목 (필수),
                "publisher": 언론사 (선택),
                "journalist": 기자명 (선택),
                "publish_date": 게재일 (선택, datetime 또는 ISO string)
            }

        Returns:
            article_id (int) 또는 None (실패 시)
        """
        if not self.client:
            return None

        try:
            url = article_data["url"]

            # 1. 기존 기사 확인
            response = self.client.table("articles").select("id").eq("url", url).execute()

            if response.data:
                article_id = response.data[0]["id"]
                logger.info(f"📌 기존 기사 사용 (ID: {article_id})")
                return article_id

            # 2. 새 기사 삽입
            # publish_date 타입 변환 (datetime → ISO string)
            publish_date = article_data.get("publish_date")
            if isinstance(publish_date, datetime):
                publish_date = publish_date.isoformat()

            new_article = {
                "url": url,
                "title": article_data["title"],
                "publisher": article_data.get("publisher"),
                "journalist": article_data.get("journalist"),
                "publish_date": publish_date
            }

            response = self.client.table("articles").insert(new_article).execute()
            article_id = response.data[0]["id"]
            logger.info(f"✨ 새 기사 저장 (ID: {article_id})")
            return article_id

        except Exception as e:
            logger.error(f"❌ DB Article Error: {e}", exc_info=True)
            return None

    async def save_analysis(
        self,
        article_id: int,
        result: Dict[str, Any],
        duration: float,
        user_id: Optional[str] = None
    ) -> Optional[int]:
        """
        분석 결과 저장

        Args:
            article_id: 기사 ID (save_or_get_article 반환값)
            result: 분석 결과 {
                "reports": {"comprehensive": ..., "journalist": ..., "student": ...},
                "detected_categories": ["1-1-1", "1-7-3", ...]
            }
            duration: 분석 소요 시간 (초)
            user_id: (선택) 로그인한 사용자 UUID

        Returns:
            analysis_result_id (int) 또는 None (실패 시)
        """
        if not self.client or not article_id:
            return None

        try:
            reports = result.get("reports", {})

            data = {
                "article_id": article_id,
                "user_id": user_id,
                "comprehensive_report": reports.get("comprehensive"),
                "journalist_report": reports.get("journalist"),
                "student_report": reports.get("student"),
                "detected_categories": result.get("detected_categories", []),
                "duration_seconds": duration,
            }

            response = self.client.table("analysis_results").insert(data).execute()
            result_id = response.data[0]["id"]
            logger.info(f"💾 분석 결과 저장 완료 (ID: {result_id})")
            return result_id

        except Exception as e:
            logger.error(f"❌ DB Analysis Error: {e}", exc_info=True)
            return None

    async def get_analysis_by_id(self, analysis_id: int) -> Optional[Dict]:
        """
        분석 ID로 결과 조회 (article 정보 포함)

        Args:
            analysis_id: 분석 결과 ID

        Returns:
            분석 결과 딕셔너리 또는 None
        """
        if not self.client:
            return None

        try:
            response = self.client.table("analysis_results") \
                .select("*, articles(*)") \
                .eq("id", analysis_id) \
                .single() \
                .execute()

            return response.data
        except Exception as e:
            logger.error(f"❌ 분석 결과 조회 실패: {e}", exc_info=True)
            return None
```

        user_id (UUID string) 또는 None (비로그인)
    """
    if not db_manager.client:
        return None

    auth_header = request.headers.get("Authorization")
    if not auth_header or not auth_header.startswith("Bearer "):
        return None

    token = auth_header.split(" ")[1]

    try:
        # Supabase JWT 검증
        user = db_manager.client.auth.get_user(token)
        return user.user.id
    except Exception as e:
        print(f"⚠️ JWT 검증 실패: {str(e)}")
        return None


@app.post("/analyze", response_model=AnalyzeResponse)
async def analyze_article(body: AnalyzeRequest, request: Request):
    """
    기사 분석 엔드포인트 (DB 저장 + 사용자 연결)

    변경사항:
    - 분석 결과를 Supabase에 자동 저장
    - 로그인한 사용자의 경우 user_id 연결
    - 저장 실패해도 사용자에게는 결과 반환 (서비스 중단 방지)
    """
    # 사용자 ID 추출 (로그인한 경우에만)
    user_id = get_user_id_from_token(request)
    if user_id:
        print(f"👤 로그인 사용자: {user_id}")
    else:
        print(f"👤 비로그인 사용자")

    try:
        # 분석 시작 시간 기록
        start_time = time.time()

        # 1. 기사 스크래핑
        print(f"📰 기사 스크래핑 시작: {body.url}")
        article_data = scraper.scrape(str(body.url))
        print(f"✅ 스크래핑 완료: {article_data['title'][:50]}...")

        # 2. 기사 분석 (2단계 파이프라인)
        print(f"🔍 기사 분석 시작...")
        result = await analyzer.analyze(article_data)
        print(f"✅ 분석 완료")

        # 분석 소요 시간 계산
        duration = time.time() - start_time

        # 3. DB에 저장 (비동기, 실패해도 결과는 반환)
        article_id = await db_manager.save_or_get_article(article_data)
        if article_id:
            await db_manager.save_analysis(
                article_id=article_id,
                result=result,
                duration=duration,
                user_id=user_id
            )

        return result

    except ValueError as e:
        # 스크래핑 또는 분석 중 발생한 에러
        raise HTTPException(status_code=400, detail=str(e))

    except Exception as e:
        # 예상치 못한 에러
        import traceback
        from datetime import datetime

        error_msg = f"[{datetime.now()}] Error processing {body.url}: {str(e)}\n{traceback.format_exc()}\n{'='*50}\n"

        try:
            with open("backend_error.log", "a", encoding="utf-8") as f:
                f.write(error_msg)
        except Exception as log_err:
            print(f"Failed to write log: {log_err}")

        print(f"❌ 오류 발생: {str(e)}")
        raise HTTPException(
            status_code=500,
            detail=f"서버 오류가 발생했습니다: {str(e)}"
        )

// frontend/app/auth/callback/route.ts
import { NextResponse } from 'next/server'
import { cookies } from 'next/headers'
import { createServerClient } from '@supabase/ssr'

export async function GET(request: Request) {
  const { searchParams, origin } = new URL(request.url)
  const code = searchParams.get('code')
  const next = searchParams.get('next') ?? '/'

  if (code) {
    const cookieStore = cookies()
    const supabase = createServerClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
      {
        cookies: {
          get(name: string) {
            return cookieStore.get(name)?.value
          },
          set(name: string, value: string, options: any) {
            cookieStore.set({ name, value, ...options })
          },
          remove(name: string, options: any) {
            cookieStore.delete({ name, ...options })
          },
        },
      }
    )

    const { error } = await supabase.auth.exchangeCodeForSession(code)

    if (!error) {
      return NextResponse.redirect(`${origin}${next}`)
    }
  }

  // 에러 발생 시 로그인 페이지로
  return NextResponse.redirect(`${origin}/login`)
}

## 8. 주의사항 및 문제 해결

### 8.1 닉네임 중복 문제

**증상**: 동시에 가입한 사용자가 같은 닉네임을 가질 수 있음

**해결**: 트리거에 중복 방지 로직 포함됨 (랜덤 접미사 추가)

**검증**:
```sql
SELECT nickname, COUNT(*) FROM profiles GROUP BY nickname HAVING COUNT(*) > 1;
```

---

### 8.2 publish_date 타입 오류

**증상**: `TypeError: Object of type datetime is not JSON serializable`

**원인**: Python datetime 객체를 JSON 직렬화할 수 없음

**해결**: `database.py`의 `save_or_get_article`에 변환 로직 추가됨
```python
if isinstance(publish_date, datetime):
    publish_date = publish_date.isoformat()
```

---

### 8.3 JWT 검증 실패

**증상**: 로그인했는데 user_id가 None

**원인**:
1. 프론트엔드에서 토큰 전달 안됨
2. 백엔드에서 검증 실패
3. 토큰 만료

**디버깅**:
```python
# main.py에 로그 추가
token = auth_header.split(" ")[1]
print(f"🔍 Token: {token[:20]}...")  # 토큰 일부만 출력
```

**해결**:
- 프론트엔드: `session.access_token` 확인
- 백엔드: `db_manager.client.auth.get_user(token)` 에러 로그 확인

---

### 8.4 RLS 정책 문제

**증상**: 프론트엔드에서 데이터 조회 불가

**원인**: RLS 정책이 너무 엄격함

**확인**:
```sql
-- Supabase SQL Editor에서 실행
SELECT * FROM analysis_results LIMIT 1;
-- 백엔드(Service Role): 성공
-- 프론트엔드(Anon Key): 정책에 따라 성공/실패
```

**해결**: "SELECT USING (true)" 정책 확인

---

### 8.5 통계 함수 호출 실패

**증상**: `function get_trending_articles(integer) does not exist`

**원인**: SQL 함수가 생성되지 않음

**확인**:
```sql
SELECT proname FROM pg_proc WHERE proname LIKE 'get_%';
```

**해결**: "1.2 Phase 3 통계 함수" SQL 재실행

---

### 8.6 Supabase 무료 티어 초과

**증상**: `Database size exceeded`

**확인**:
```sql
SELECT pg_size_pretty(pg_database_size('postgres'));
```

**해결 옵션**:
1. 오래된 분석 결과 삭제 (3개월 이상)
2. 리포트 텍스트 압축
3. 유료 플랜 전환 ($25/month)
